OOP №1

class sup(object):
    def __init__(self):
        print('sup')



class high(sup):
    def __init__(self):
        print('high')

        super(high, self).__init__()


class low(high):
    def __init__(self):
        print('-'*20)
        super(low, self).__init__()


        print('low')
        mro = type(self).mro()
        print('mro low: ' + str(mro))

if __name__ == "__main__":
    l = low()
